server:
  port: 8080
  address: 0.0.0.0

spring:
  cloud:
    gateway:
      # Global CORS - Vite + React için
      globalcors:
        corsConfigurations:
          '[/**]':
            allowedOrigins:
              - "http://localhost:5173"
            allowedMethods:
              - GET
              - POST
              - PUT
              - PATCH
              - DELETE
              - OPTIONS
            allowedHeaders:
              - "*"
              - "Authorization"
              - "Content-Type"
              - "X-Requested-With"
              - "Accept"
            exposedHeaders:
              - "Authorization"
              - "X-Total-Count"
              - "Content-Disposition"
            allowCredentials: true
            maxAge: 3600
        add-to-simple-url-handler-mapping: true

      # Routes
      routes:
        # ========== ANALYTICS SERVICE (ÖNCELİKLİ!) ==========
        # ✅ TÜM analytics route'larını TEK route'a indirgedik
        - id: analytics-service
          uri: lb://analytics-service
          predicates:
            - Path=/api/analytics/**
          filters:
            - RewritePath=/api/analytics/(?<segment>.*), /api/v1/analytics/$\{segment}
          metadata:
            auth-required: "false"
          order: 1  # ✅ EN YÜKSEK ÖNCELİK

        # ========== CUSTOMER SERVICE ==========
        - id: customer-service
          uri: lb://customer-service
          predicates:
            - Path=/api/customers/**
          metadata:
            auth-required: "true"
            roles: "USER,ADMIN"
          order: 2

        # ========== DISCOVERY SERVER ==========
        - id: discovery-server
          uri: http://localhost:8761
          predicates:
            - Path=/eureka/web
          filters:
            - SetPath=/
          metadata:
            auth-required: "false"
          order: 10

        - id: discovery-server-static
          uri: http://localhost:8761
          predicates:
            - Path=/eureka/**
          metadata:
            auth-required: "false"
          order: 11

      # Default Filters
      default-filters:
        - name: Retry
          args:
            retries: 3
            statuses: BAD_GATEWAY,GATEWAY_TIMEOUT
            methods: GET,POST,PUT,PATCH,DELETE
            backoff:
              firstBackoff: 50ms
              maxBackoff: 500ms
              factor: 2
              basedOnPreviousValue: true

# Eureka
eureka:
  client:
    service-url:
      defaultZone: http://localhost:8761/eureka/
    fetch-registry: true
    register-with-eureka: true
  instance:
    prefer-ip-address: true
    instance-id: ${spring.application.name}:${server.port}
    metadata-map:
      public: "true"
      security-level: "gateway"

# Actuator
management:
  endpoints:
    web:
      exposure:
        include: health,info,gateway,routes,metrics
  endpoint:
    health:
      show-details: always
    gateway:
      enabled: true

# Logging
logging:
  level:
    org.springframework.cloud.gateway: DEBUG  # ✅ DEBUG yap
    reactor.netty.http.client: DEBUG
    com.sosyalmedia.apigateway: DEBUG
  pattern:
    console: "%d{yyyy-MM-dd HH:mm:ss} - %msg%n"
